74static void Medir_task()
75{
76 while (true)
77 {
78 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
79 if (on_off_pantalla)
80 {
81 distancia = HcSr04ReadDistanceInCentimeters();
82
83 if (distancia <= 10)
84 {
85 LedOff(LED_1);
86 LedOff(LED_2);
87 LedOff(LED_3);
88 }
89 else if (distancia > 10 && distancia <= 20)
90 {
91 LedOn(LED_1);
92 LedOff(LED_2);
93 LedOff(LED_3);
94 }
95 else if (distancia > 20 && distancia <= 30)
96 {
97 LedOn(LED_1);
98 LedOn(LED_2);
99 LedOff(LED_3);
100 }
101 else
102 {
103 LedOn(LED_1);
104 LedOn(LED_2);
105 LedOn(LED_3);
106 }
107 if (!hold)
108 {
109 LcdItsE0803Write(distancia);
110 }
111 }
112 else
113 {
114 LcdItsE0803Off();
115 LedsOffAll();
116 }
117 }
118}
